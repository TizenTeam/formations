Imprimante
	encreRestante: int
	
	Imprimante( capaciteEncre: int )

Verre

	Verre( contenance: int )

	// éventuellement ca déborde

Le programme principal

Verre v = new Verre( 50 );
v.ajouter( 150 );
v.ajouter( 150 );

Exercice Voiture 1

Car
	engine: Engine
	
	turnKeyOn(): void // affiche 'la voiture démarre' puis démarre le moteur
	turnKeyOff(): void // affiche 'la voiture 'éteint' puis éteint le moteur

Engine
	started: boolean
	
	start(): void // affiche 'Le moteur démarre' ou 'CRRRR' s'il est déjà démarré
	stop(): void // affiche 'Le moteur s'arrête' ou 'Moteur déjà éteint' s'il est déjà éteint

Programme principal :

	Car car = new Car();
	
	car.turnKeyOn();
	car.turnKeyOn();
	
	car.turnKeyOff();
	car.turnKeyOff();
